<script>
const scenes = document.querySelectorAll(".scene");
const images = document.querySelectorAll(".bg");

let i = 0;

function nextScene() {
  scenes[i].classList.remove("active");
  images[i % images.length].classList.remove("show");

  i++;

  scenes[i % scenes.length].classList.add("active");
  images[i % images.length].classList.add("show");
}

setInterval(nextScene, 6000); // ~30 sec

// hearts
function heart() {
  let h = document.createElement("div");
  h.className = "heart";
  h.innerHTML = "❤️";
  h.style.left = Math.random()*100 + "vw";
  h.style.fontSize = (Math.random()*20+10)+"px";
  document.body.appendChild(h);
  setTimeout(()=>h.remove(),6000);
}
setInterval(heart,300);
